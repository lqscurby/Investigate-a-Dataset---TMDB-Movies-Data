#define the function
def calculate(column):
    #high profit
    high=tmdb_data[column].idxmax()
    high_data=pd.DataFrame(tmdb_data.loc[high])
    
    #low profit
    low=tmdb_data[column].idxmin()
    low_data=pd.DataFrame(tmdb_data.loc[low])
    
    #collect data
    data=pd.concat([high_data, low_data], axis=1)
    
    return data

#genres have more than one category and have string that holds multiple values. We need to separate the string.
#create a column for each genre - 1 if genre is in genres else 0

def genre_data(column):
#create a column and separate the string by '|'   
    genre_data = tmdb_data[column].str.cat(sep = '|')
#use pandas series and store the values separately
    genre_data = pd.Series(genre_data.split('|'))
#arranging order to look clean
    count = genre_data.value_counts(ascending = False)
    
    return count

#variable to store the returned values
count = genre_data('genres')
